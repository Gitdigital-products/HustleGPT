hustle_gpt.py
# Hustle GPT - Business Idea Generator
import openai
import json

class HustleGPT:
    def __init__(self, api_key):
        openai.api_key = api_key
        self.context = "You are an expert entrepreneurship coach..."
    
    def generate_business_idea(self, niche, budget):
        prompt = f"Generate a business idea in the {niche} niche with a budget of ${budget}."
        
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": self.context},
                {"role": "user", "content": prompt}
            ],
            temperature=0.8,
            max_tokens=500
        )
        
        return response.choices[0].message.content

# Initialize Hustle GPT
hustle_bot = HustleGPT("your-api-key-here")

# Generate a business idea
idea = hustle_bot.generate_business_idea("sustainable products", 5000)
